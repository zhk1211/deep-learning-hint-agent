def good(x):
    return x % 2 > 0 and x % 3 > 0 and x % 5 > 0 and x % 7 > 0
 
def get_naive(x):
    ans = 0
    for i in range(x):
        if good(i):
            ans += 1
    return ans
 
def get(r):
    return (r // 210) * get_naive(210) + get_naive(r % 210)
 
t = int(input())
for i in range(t):
    l, r = map(int, input().split())
    print(get(r+1) - get(l))

        res = 0
        ps = [2, 3, 5, 7]

            m = 1

            for i in range(4):
                if (mask >> i) & 1:

2125D - Segments Covering

Idea: BledDest

Tutorial
Tutorial is loading...
Solution 1 (Neon)
#include <bits/stdc++.h>
 
using namespace std;
 
using pt = pair<int, int>;
 
const int MOD = 998244353;
 
int add(int x, int y) {
   x += y;
   if (x >= MOD) x -= MOD;
   if (x < 0) x += MOD;
   return x;
}
 
int mul(int x, int y) {
  return x * 1LL * y % MOD;
}
 
int binpow(int x, int y) {
  int z = 1;
  while (y) {
    if (y & 1) z = mul(z, x);
    x = mul(x, x);
    y >>= 1;
  }
  return z;
}
 
int divide(int x, int y) {
  return mul(x, binpow(y, MOD - 2));
}
 
int main() {
  ios::sync_with_stdio(false); cin.tie(0);
  int n, m;
  cin >> n >> m;
  vector<vector<pt>> a(m + 1);
  for (int i = 0; i < n; ++i) {
    int l, r, p, q;
    cin >> l >> r >> p >> q;
    a[r].emplace_back(l - 1, divide(p, q));
  }
  vector<int> pr(m + 1);
  pr[0] = 1;
  for (int i = 1; i <= m; ++i) {
    int cur = 1;
    for (auto [_, p] : a[i]) cur = mul(cur, add(1, -p));
    pr[i] = mul(pr[i - 1], cur);
  }
  vector<int> dp(m + 1);
  dp[0] = 1;
  for (int i = 1; i <= m; ++i) {
    for (auto [l, p] : a[i]) { 
      int cur = divide(pr[i], pr[l]);
      cur = divide(cur, add(1, -p));
      cur = mul(cur, p);
      dp[i] = add(dp[i], mul(dp[l], cur));
    }
  }
  cout << dp[m] << endl;  
}

2125E - Sets of Complementary Sums

Idea: FelixArg

Tutorial
Tutorial is loading...
Solution (FelixArg)
#include<bits/stdc++.h>
using namespace std;
 
#define int long long
 
const int MOD = 998'244'353;
 
void solve(){
    int n, x;
    cin >> n >> x;
    // need sum(mx + 1 - ai) <= mx + 1
    // start with 1, 2, 3, ... n
    // min sum + 1 is n + 1
 
    if (n == 1){
        cout << x << '\n';
        return;
    }
 
    int s_cur = n * (n + 1) / 2;
    int need = s_cur - (n + 1);
 
    // leads to n is O(sqrt(s))
    if (n + need > x){
        cout << 0 << '\n';
        return;
    }
 
    // so we can do dp[sum] knapsack
 
    vector<int> dp(x + 1);
    dp[0] = 1;
 
    // for -1 and 0
    for (int j = 0; j < 2; j++){
        for (int i = 0; i < x; i++){
            dp[i + 1] = (dp[i + 1] + dp[i]) % MOD;
        }
    }
 
    for (int i = 1; i <= n - 2; i++){
        for (int j = 0; j < x; j++){
            // increase sum by i -> mx increase by 1 + i
            if (j + i + 1 <= x){
                dp[j + i + 1] = (dp[j + i + 1] + dp[j]) % MOD;
            }
        }
    }
 
    int ans = 0;
    for (int i = 0; i <= x; i++){
        int cur = n + need + i;
        if (cur > x){
            break;
        }
        ans = (ans + dp[i]) % MOD;
    }
 
    cout << ans << '\n';
}
 
signed main()
{
#ifdef FELIX
    auto _clock_start = chrono::high_resolution_clock::now();
#endif
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);
 
    int t = 1;
    cin >> t;
    while(t--){
        solve();
    }
 
#ifdef FELIX
    cerr << "Executed in " << chrono::duration_cast<chrono::milliseconds>(
        chrono::high_resolution_clock::now()
            - _clock_start).count() << "ms." << endl;
#endif
    return 0;
}

2125F - Timofey and Docker

Idea: FelixArg

Tutorial
Tutorial is loading...
Solution (awoo)
#include <bits/stdc++.h>
 
using namespace std;
 
#define forn(i, n) for(int i = 0; i < int(n); i++) 
 
const int INF = 1e8;
const string al = "docker";
 
int main(){
    cin.tie(0);
    ios::sync_with_stdio(false);
    int tc;
    cin >> tc;
    while (tc--){
        string s;
        cin >> s;
        int n = s.size();
        int mx = n / 6;
        int m;
        cin >> m;
        vector<int> dx(mx + 2);
        forn(i, m){
            int l, r;
            cin >> l >> r;
            r = min(r, mx);
            if (l > r) continue;
            ++dx[l];
            --dx[r + 1];
        }
        forn(i, mx + 1) dx[i + 1] += dx[i];
 
        int cur = 0;
        vector<int> dif(max(0, n - 5));
        forn(i, n - 5){
            forn(j, 6)
                dif[i] += s[i + j] != al[j];
            cur += dif[i] == 0;
        }
 
        int mxcnt = *max_element(dx.begin(), dx.end());
        int k = -1;
        int ans = INF;
        forn(i, mx + 1){
            if (dx[i] != mxcnt) continue;
            if (i <= cur){
                ans = min(ans, cur - i);
                continue;
            }
            k = i;
            break;
        }
 
        if (k == -1){
            cout << ans << '\n';
            continue;
        }
 
        auto calc = [&](int d){
            vector<pair<long long, int>> dp(n + 1, {(long long)1e18, -1});
            dp[0] = {0, 0};
            forn(i, n){
                dp[i + 1] = min(dp[i + 1], dp[i]);
                if (i + 6 <= n)
                    dp[i + 6] = min(dp[i + 6], {dp[i].first + dif[i] + d, dp[i].second - 1});
            }
            return dp[n];
        };
 
        int l = -INF, r = 0, sv = INF;
        while (l <= r){
            int m = (l + r) / 2;
            auto [val, cnt] = calc(m);
            if (-cnt >= k){
                sv = val - k * 1ll * m;
                l = m + 1;
            }
            else{
                r = m - 1;
            }
        }
        ans = min(ans, sv);
 
        cout << ans << '\n';
    } 
}

Slightly different derivation of the sufficient and necessary condition for question E. Consider any (sorted) output set x = {x(1),x(2),...,x(n)} that we want to make a input set for for. Consider the difference array {d(2),d(3),..d(n)} where d(i) = x(i)-x(i-1) we know there must be some term z after which the following terms come, z+d(n), z+d(n)+d(n-1), z+d(n)+d(n-1)+d(n-2).., in the input set. Thus let us only have those terms for now. Note that it is optimal to make z=1, since any higher z would only make the input sum higher raising all our output terms by a constant, which we can just do instead by adding more 1 terms should we need to. Now we have the input array {1,1+d(n),1+d(n)+d(n-1),...,1+(sum of all d(i)}. Lets find the sum of this array, note that 1 adds to n terms, (x(n)-x(n-1)) adds to n-1 terms, (x(n-1)-x(n-2)) adds to n-3 terms, etc. It is not hard to see that each x term will be subtracted once except for x(n), thus the final sum is just n+x(n)*(n-1)-(sum of all x(i) s.t i<n). Now, from the construction, we know that this corresponds to an output array that is off by the original set x by some constant, i.e it has the same difference array when sorted. The only question now is, is the highest term, and thus all the terms, small enough? Remember that too small is ok since we can raise the sum, but too big is not ok. We know that the largest term will be equal to the sum-1, since the lowest term is 1, so we know it is just n+x(n)*(n-1)-(sum of all x(i) s.t i<n)-1 and this must be less than or equal to x(n). Thus we have n+x(n)*(n-1)-(sum of all x(i) s.t i<n) — 1 <= x(n). Rearranging this we get x(n)*(n-1)+(n-1) — (sum of all x(i) s.t i<n) <= x(n) and then we get (x(n)+1)*(n-1) — (sum of all x(i)) <= 0, exactly the condition found in the editorial.

In the editorial for E, is $$$\mathrm{dp}[st]$$$ really equal to 1? From what I understand we should have $$$\mathrm{dp}[st+n] = 1$$$ instead (with $$$\mathrm{dp}[st]=0$$$).

$$$\mathrm{dp}[i]$$$ is defined as "the number of different sets of complementary sums such that the maximum in these sets equals $$$i$$$ and equality in condition (1) holds". As $$$st$$$ is defined as the number of additions of ones to all of the elements of the set {$$$1, 2, \dots, n$$$} needed to make it satisfy the condition (1), it can be zero (with $$$n=2$$$). With $$$n=2$$$, $$$\mathrm{dp}[0] \neq 1$$$, because every set of complementary sums with more than one element* has only positive values inside, so the maximum of those values can't be zero. On the other hand, $$$\mathrm{dp}[st+n] = 1$$$ makes sense -- the set {$$$1, 2, \dots, n$$$} after the additions will have its maximum equal to $$$n+st$$$, thus $$$\mathrm{dp}[n+st] = 1$$$.

*in the editorial it is not specified that in the $$$\mathrm{dp}$$$ we only care about sets of size $$$n$$$, but I don't see how that wouldn't be true.
